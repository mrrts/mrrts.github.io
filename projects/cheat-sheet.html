Freezing a variable

Use an immediately invoked function expression (IIFE) that returns the function with the current value for the variable


var my_obj = {}
for (var i = 0; i < 4; i++) {
	my_obj['my_method' + String(i)] = (function (val) {
		return function () {
			console.log('The current value of i in the for-loop is', val)
		}
	})(i);
}

my_obj.my_method0();
// ->  "The current value of i in the for-loop is 0"

my_obj.my_method1();
// ->  "The current value of i in the for-loop is 1"
